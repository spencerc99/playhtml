    <TabItem label="Vanilla HTML">
```html
<script type="module">
  import { playhtml } from "https://unpkg.com/playhtml";
  playhtml.init();
</script>
<img can-toggle id="lamp" src="/noguchi-hanging-lamp.png" />
```
    </TabItem>
    <TabItem label="React">
```tsx
// npm install @playhtml/react @playhtml/common
import { PlayProvider, CanToggleElement } from "@playhtml/react";

<PlayProvider>
  <CanToggleElement>
    <img id="lamp" src="/noguchi-hanging-lamp.png" />
  </CanToggleElement>
</PlayProvider>
```
